# A garrison captain is renowned for their ability, charisma, and popularity
VIETmisc.3002 = {
	type = character_event
	title = VIETmisc.3002.t
	desc = VIETmisc.3002.desc
	theme = VIET_theme_small_town
	
	left_portrait = root
	right_portrait = {
		character = scope:captain
		animation = marshal
	}
	
    weight_multiplier = {
        base = 0
		modifier = {
			add = {
				every_held_title = {
					limit = {
						tier = tier_county					
						is_landless_type_title = no
					}
					add = 1
				}
			}		
		}		
    }
	
	immediate = {
		random_held_title = {
			limit = {
				tier = tier_county					
				is_landless_type_title = no
			}	
			save_scope_as = county
		}
		create_character = {
			location = root.capital_province
			template = VIET_popular_captain_template
			save_scope_as = captain
		}	
	}
	
	option = { # Invite to court
		name = VIETmisc.3002.a
		trigger = {
			OR = {
				gold >= 25
				is_ai = no
			}
		}
		remove_short_term_gold = 25
		add_courtier = scope:captain
		if = {
			limit = {
				can_add_hook = {
					target = scope:captain
					type = favor_hook
				}
			}
			add_hook = {
				target = scope:captain
				type = favor_hook
			}
		}
		add_character_modifier = {
			modifier = VIET_promoted_beloved_guard_captain
			years = 10
		}
	}	
	option = { # Let the guard captain stay where they are
		name = VIETmisc.3002.b
		scope:county = {
			add_county_modifier = {
				modifier = VIET_beloved_guard_captain
				years = 15
			}		
		}
		hidden_effect = {
			scope:captain = {		
				death = {
					death_reason = death_disappearance
				}
			}			
		}
	}	
	option = { # Make the guard captain ruler of that county
		name = VIETmisc.3002.c
		trigger = {
			is_ai = no
			any_held_title = {
				is_landless_type_title = no
				count >= 2 #Don't give away your only title
			}
			highest_held_title_tier >= tier_duchy
		}
		# Assign title to captain
		create_title_and_vassal_change = {
			type = granted
			save_scope_as = change
			add_claim_on_loss = yes
		}
		scope:county = {
			change_title_holder = {
				holder = scope:captain
				change = scope:change
			}		
		}
		resolve_title_and_vassal_change = scope:change
		# Gain a strong hook against the captain.
		reverse_add_opinion = {
			modifier = grateful_opinion
			target = scope:captain
			opinion = 50
		}
		add_hook = {
			type = loyalty_hook
			target = scope:captain
		}
		#scope:captain = {
		#	get_title = scope:county	
		#}
		add_character_modifier = {
			modifier = VIET_ennobled_beloved_guard_captain
			years = 10
		}
	}
}

# Red billed queleas infestation - "feathered locusts" of Africa
VIETmisc.3028 = {
	type = character_event
	title = VIETmisc.3028.t
	desc = VIETmisc.3028.desc
	theme = unfriendly
	override_background = { reference = VIET_background_small_town }
	
	left_portrait = {
		character = root
		animation = worry
	}
	
	trigger = {
		NOT = { has_game_rule = VIET_conversion_universe_events }
		any_held_title = {
			tier = tier_county
			is_landless_type_title = no
			any_county_province = {
				OR = {
					geographical_region = ghw_region_ethiopia
					geographical_region = custom_sahel_proper
				}		
			}
		}	
	}
	
    weight_multiplier = {
        base = 0
		modifier = {
			add = {
				every_held_title = {
					limit = {
						tier = tier_county					
						is_landless_type_title = no
					}
					add = 1
				}
			}		
		}		
    }
	
	immediate = {
		random_held_title = {
			limit = {
				tier = tier_county
				any_county_province = {
					OR = {
						geographical_region = ghw_region_ethiopia
						geographical_region = custom_sahel_proper
					}			
				}
			}	
			save_scope_as = county
		}
	}
	
	option = {
		name = VIETmisc.3028.a
		scope:county = {
			add_county_modifier = {
				modifier = VIET_feathered_locusts
				years = 5
			}		
		}
		if = {
			limit = {
				has_game_rule = VIET_historical_context_on
			}
			custom_tooltip = VIETmisc.3028.info
		}
	}
	
	option = {
		name = VIETmisc.3028.b
		trigger = {
			stewardship >= decent_skill_rating			
		}
		skill = stewardship
		add_gold = 30
		VIET_massive_prestige_loss_effect = yes
		if = {
			limit = {
				has_game_rule = VIET_historical_context_on
			}
			custom_tooltip = VIETmisc.3028.info
		}
	}

}
