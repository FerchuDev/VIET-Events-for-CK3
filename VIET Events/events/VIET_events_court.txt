			}

			add_opinion = {
				target = root

		}

			}	
			scope:chaplain = {
				add_opinion = {
					target = root
					modifier = VIET_opinion_agrees_intellectual
				}
			}	
		}

				has_character_flag = VIET_intellectual_event_prefer_traditionalist
			}	
			scope:chaplain = {
				add_opinion = {
					target = root

			}
